\text{Length of straw, $L = \frac{\lambda}{2}$}\\

\text{$L = (4.20 \pm 0.05)\,\text{cm} = (4.20\times10^{-2} \pm 5\times10^{-4})\,\text{m}$}\\
\text{$\lambda = 2L = (8.4\times10^{-2} \pm 1\times10^{-3})\,\text{m}$}\\

\text{let $y = \frac{1}{\lambda} = \lambda^{-1}$}\\
\text{$\frac{\Delta y}{y} = \left| \frac{\Delta \lambda}{\lambda} \right|$}\\
\text{$\Delta y = y \cdot \left| \frac{\Delta \lambda}{\lambda} \right| = \frac{1}{\lambda} \cdot \left| \frac{\Delta \lambda}{\lambda} \right|$}\\

\text{$y = \frac{1}{8.4\times10^{-2}} = 11.90476\,\text{m}^{-1}$}\\
\text{$\Delta y = 11.90476 \cdot \left| \frac{1\times10^{-3}}{8.4\times10^{-2}} \right| = 0.14172\,\text{m}^{-1} = 0.1\text{m}^{-1}$ to 1 s.f.}\\

\text{Taking into account uncertainties, $\frac{1}{\lambda} = (11.9 \pm 0.1)\,\text{m}^{-1}$}

\hrulefill
